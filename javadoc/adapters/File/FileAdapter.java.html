3    import java.io.File;

5    /**
6     * Adapter class for file-related operations.
7     */
8    public class FileAdapter {
9        private static final String BASE_PATH = "src/resources/maps/";
10   
11       /**
12        * Checks if a file exists in the specified directory.
13        *
14        * @param p_fileName The name of the file to check.
15        * @return The File object if it exists, otherwise null.
16        */
17       public static File isFileExists(String p_fileName) {
18           String l_fileName = p_fileName.endsWith(".map") ? p_fileName : p_fileName + ".map";
19           File l_file = new File(BASE_PATH + l_fileName);
20           if (l_file.exists()) return l_file;
21           return null;
22       }
23   }
